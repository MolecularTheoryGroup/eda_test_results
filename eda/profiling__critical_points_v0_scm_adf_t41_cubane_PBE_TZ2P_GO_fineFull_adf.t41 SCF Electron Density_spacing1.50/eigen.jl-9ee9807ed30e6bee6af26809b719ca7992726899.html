137 if ishermitian(A)

159 @inline function _eig(::Size{(2,2)}, A::LinearAlgebra.RealHermSymComplexHerm{T}, permute, scale) where {T <: Real}

204 # [This] is an implementation of Algorithm 8.2.3 (Symmetric QR Algorithm) described in
205 # Matrix Computations,2nd edition, by G. H. Golub and C. F. Van Loan, The Johns Hopkins
206 # University Press, Baltimore MD, Fourth Printing 1993. Algorithm 8.2.1 (Householder
207 # Tridiagonalization) is used to reduce matrix A to tridiagonal D′. Algorithm 8.2.2
208 # (Implicit Symmetric QR Step with Wilkinson Shift) is used for the iterative reduction
209 # from tridiagonal to diagonal. Numerically, we have errors E=RTAR−D. Algorithm 8.2.3
210 # mentions that one expects |E| is approximately μ|A|, where |M| denotes the Frobenius norm
211 # of M and where μ is the unit roundoff for the floating-point arithmetic: 2−23 for float,
212 # which is FLTEPSILON = 1.192092896e-7f, and 2−52 for double, which is
213 # DBLEPSILON = 2.2204460492503131e-16.

226 function get_cos_sin(u::T,v::T) where {T}
227 max_abs = max(abs(u), abs(v))
228 if max_abs > 0
229 u,v = (u,v) ./ max_abs
230 len = sqrt(u^2 + v^2)
231 cs, sn = (u,v) ./ len
232 if cs > 0
233 cs = -cs
234 sn = -sn
235 end
236 T(cs), T(sn)
237 else
238 T(-1), T(0)
239 end
240 end

251 # Givens reflections
252 update0(Q, c, s) = Q * @SMatrix [c 0 -s; s 0 c; 0 1 0]
253 update1(Q, c, s) = Q * @SMatrix [0 1 0; c 0 s; -s 0 c]
254 # Householder reflections
255 update2(Q, c, s) = Q * @SMatrix [c s 0; s -c 0; 0 0 1]
256 update3(Q, c, s) = Q * @SMatrix [1 0 0; 0 c s; 0 s -c]

258 is_rotation = false
259
260 # If `aggressive` is `true`, the iterations occur until a superdiagonal
261 # entry is exactly zero, otherwise they occur until it is effectively zero
262 # compared to the magnitude of its diagonal neighbors. Generally the non-
263 # aggressive convergence is acceptable.
264 #
265 # Even with `aggressive = true` this method is faster than the one it
266 # replaces and in order to keep the old interface, aggressive is set to true
267 aggressive = true

269 # the input is symmetric, so we only consider the unique elements:
270 a00, a01, a02, a11, a12, a22 = A[1,1], A[1,2], A[1,3], A[2,2], A[2,3], A[3,3]

281 b01 = s * term0 - c * term1
282 term0 = s * a00 - c * a01
283 term1 = s * a01 - c * a11
284 b11 = s * term0 - c * term1
285 b12 = s * a02 - c * a12

288 # Givens reflections, B' = G^T * B * G, preserve tridiagonal matrices
289 max_iteration = 2 * (1 + precision(T) - exponent(floatmin(T)))
290
291 if abs(b12) <= abs(b01)
292 saveB00, saveB01, saveB11 = b00, b01, b11
293 for iteration in 1:max_iteration
294 # compute the Givens reflection
295 c2, s2 = get_cos_sin((b00 - b11) / 2, b01)
296 s = sqrt((1 - c2) / 2)
297 c = s2 / 2s
298
299 # update Q by the Givens reflection
300 Q = update0(Q, c, s)
301 is_rotation = !is_rotation
302
303 # update B ← Q^T * B * Q, ensuring that b02 is zero and |b12| has
304 # strictly decreased

306 term0 = c * saveB00 + s * saveB01
307 term1 = c * saveB01 + s * saveB11
308 b00 = c * term0 + s * term1
309 b11 = b22

333 b11 = s * term0 - c * term1
