47 function dogleg!(p, p_c, p_i,
48 r, d, J, delta::Real)
49 T = eltype(d)
50 try
51 copyto!(p_i, J \ vec(r)) # Gauss-Newton step
52 catch e
53 if isa(e, LAPACKException) || isa(e, SingularException)
54 # If Jacobian is singular, compute a least-squares solution to J*x+r=0
55 U, S, V = svd(convert(Matrix{T}, J)) # Convert to full matrix because sparse SVD not implemented as of Julia 0.3
56 k = sum(S .> eps())
57 mrinv = V * Matrix(Diagonal([1 ./ S[1:k]; zeros(eltype(S), length(S)-k)])) * U' # Moore-Penrose generalized inverse of J
58 vecpi = vec(p_i)
59 mul!(vecpi,mrinv,vec(r))
60 else
61 throw(e)
62 end
63 end
64 rmul!(p_i, -one(T))
65
66 # Test if Gauss-Newton step is within the region
67 if wnorm(d, p_i) <= delta
68 copyto!(p, p_i) # accepts equation 4.13 from N&W for this step
69 else
70 # For intermediate we will use the output array `p` as a buffer to hold
71 # the gradient. To make it easy to remember which variable that array
72 # is representing we make g an alias to p and use g when we want the
73 # gradient
74
75 # compute g = J'r ./ (d .^ 2)
76 g = p
77 mul!(vec(g), transpose(J), vec(r))
78 g .= g ./ d.^2
79
80 # compute Cauchy point
81 p_c .= -wnorm(d, g)^2 / sum(abs2, J*vec(g)) .* g
82
83 if wnorm(d, p_c) >= delta
84 # Cauchy point is out of the region, take the largest step along
85 # gradient direction
86 rmul!(g, -delta/wnorm(d, g))
87
88 # now we want to set p = g, but that is already true, so we're done
89
90 else
91 # from this point on we will only need p_i in the term p_i-p_c.
92 # so we reuse the vector p_i by computing p_i = p_i - p_c and then
93 # just so we aren't confused we name that p_diff
94 p_i .-= p_c
95 p_diff = p_i
96
97 # Compute the optimal point on dogleg path
98 b = 2 * wdot(d, p_c, d, p_diff)
99 a = wnorm(d, p_diff)^2
100 tau = (-b + sqrt(b^2 - 4a*(wnorm(d, p_c)^2 - delta^2)))/(2a)
101 p_c .+= tau .* p_diff
102 copyto!(p, p_c)
103 end
104 end
