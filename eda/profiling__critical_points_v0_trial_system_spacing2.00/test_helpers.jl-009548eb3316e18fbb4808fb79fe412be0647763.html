34 function unzip(file::String, exdir="")::Nothing
35 @assert isfile(file) "File does not exist: $file"
36 fileFullPath = isabspath(file) ? file : joinpath(pwd(), file)
37 basePath = dirname(fileFullPath)
38 outPath = (exdir == "" ? basePath : (isabspath(exdir) ? exdir : joinpath(pwd(), exdir)))
39 isdir(outPath) ? "" : mkdir(outPath)
40 zarchive = ZipFile.Reader(fileFullPath)
41 for f in zarchive.files
42 fullFilePath = joinpath(outPath, f.name)
43 if (endswith(f.name, "/") || endswith(f.name, "\\"))
44 mkdir(fullFilePath)
45 else
46 try
47 write(fullFilePath, read(f))
48 catch e
49 @debug "Error writing file in unzip(): $e"
50 end
51 end
52 end
53 return close(zarchive)
54 return nothing
55 end

60 The `with_zip_file` function is a utility function that takes a file path and a function as inputs. If the unzipping is successful, it calls the provided function with the original file path as an argument. If any error occurs during the unzipping or function execution, the error is rethrown. Finally, it removes the original file using the `rm` function.

The function `process_file` can perform any desired operations on the unzipped file.

88 function with_zip_file(f_path::AbstractString, func::Function)
89 z_path = f_path * ".zip"
90 out = nothing
91 if isfile(z_path)
92 try
93 unzip(z_path)
94 out = func(f_path)
95 catch e
96 rethrow(e)
97 finally
98 for f in readdir(dirname(f_path))
99 if startswith(f, basename(f_path)) && !endswith(f, ".zip")
100 rm(joinpath(dirname(f_path), f))
101 end
102 end
103 end
104 else
105 if isfile(f_path)
106 out = func(f_path)
107 else
108 error("File does not exist: $f_path")
109 end
