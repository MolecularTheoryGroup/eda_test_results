300 anderson(z, cache)
301
302 Return the next iterate of the fixed-point iteration `z = g(z)` by performing Anderson
303 acceleration based on the current iterate `z` and the settings and history in the `cache`.
304 """
305 @muladd function anderson(z, cache)
306 @unpack dz, Δz₊s, z₊old, dzold, R, Q, γs, history, droptol = cache
307
308 # increase size of history
309 history += 1
310
311 # remove oldest history if maximum size is exceeded
312 max_history = length(Δz₊s)
313 if history > max_history
314 # circularly shift differences of z₊
315 for i in 1:(max_history - 1)
316 Δz₊s[i] = Δz₊s[i + 1]
317 end
318
319 # delete left-most column of QR decomposition
320 qrdelete!(Q, R, max_history)
321
322 # update size of history
323 history = max_history
324 end
325
326 # update history of differences of z₊
327 Δz₊s[history] = @.. broadcast=false z-z₊old
328
329 # replace/add difference of residuals as right-most column to QR decomposition
330 qradd!(Q, R, _vec(dz .- dzold), history)
331
332 # update cached values
333 cache.dzold = dz
334 cache.z₊old = z
335
336 # define current Q and R matrices

338
339 # check condition (TODO: incremental estimation)
340 if droptol !== nothing
341 while cond(R) > droptol && history > 1
342 qrdelete!(Q, R, history)
343 history -= 1
344 Qcur, Rcur = view(Q, :, 1:history),

346 end
347 end

350 cache.history = history

353 γscur = view(γs, 1:history)
354 ldiv!(Rcur, mul!(γscur, Qcur', _vec(dz)))

468 """
469 qrdelete!(Q, R, k)
470
471 Delete the left-most column of F = Q[:, 1:k] * R[1:k, 1:k] by updating Q and R.
472 Only Q[:, 1:(k-1)] and R[1:(k-1), 1:(k-1)] are valid on exit.
473 """
474 function qrdelete!(Q::AbstractMatrix, R::AbstractMatrix, k::Int)
475 n, m = size(Q)
476 m == LinearAlgebra.checksquare(R) || throw(DimensionMismatch())
477 1 ≤ k ≤ m || throw(ArgumentError())
478
479 # apply Givens rotations
480 for i in 2:k
481 g = first(givens(R, i - 1, i, i))
482 lmul!(g, R)
483 rmul!(Q, g')
484 end
485
486 # move columns of R
487 @inbounds for j in 1:(k - 1)
488 for i in 1:(k - 1)
489 R[i, j] = R[i, j + 1]
490 end
491 end
492
493 Q, R
494 end

496 """
497 qradd!(Q, R, v, k)
498
499 Replace the right-most column of F = Q[:, 1:k] * R[1:k, 1:k] with v by updating Q and R.
500 This implementation modifies vector v as well. Only Q[:, 1:k] and R[1:k, 1:k] are valid on
501 exit.
502 """
503 function qradd!(Q::AbstractMatrix, R::AbstractMatrix, v::AbstractVector, k::Int)
504 n, m = size(Q)
505 n == length(v) || throw(DimensionMismatch())
506 m == LinearAlgebra.checksquare(R) || throw(DimensionMismatch())
507 1 ≤ k ≤ m || throw(ArgumentError())
508
509 @inbounds for i in 1:(k - 1)
510 q = view(Q, :, i)
511 r = dot(q, v)
512
513 R[i, k] = r
514 axpy!(-r, q, v)
515 end
516
517 @inbounds begin
518 d = norm(v)
519 R[k, k] = d
520 @.. broadcast=false @view(Q[:, k])=v / d
521 end
522
523 Q, R
524 end
