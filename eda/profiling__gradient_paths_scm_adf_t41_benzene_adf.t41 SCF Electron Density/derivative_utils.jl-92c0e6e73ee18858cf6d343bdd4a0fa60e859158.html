161
162 ```math
163 W = MM - \\gamma J
164 ```
165
166 or, if `transform=true`:
167
168 ```math
169 W = \\frac{1}{\\gamma}MM - J
170 ```
171
172 where `MM` is the mass matrix (a regular `AbstractMatrix` or a `UniformScaling`),
173 `γ` is a real number proportional to the time step, and `J` is the Jacobian

706 end
707 if isnewton(nlsolver)
708 set_new_W!(nlsolver, new_W)
709 if new_jac && isdae

711 elseif new_W && !isdae

714 end

716 new_W && (integrator.stats.nw += 1)

720 @noinline function calc_W(integrator, nlsolver, dtgamma, repeat_step, W_transform = false)
721 @unpack t, uprev, p, f = integrator
722
723 next_step = is_always_new(nlsolver)
724 if next_step
725 t = t + integrator.dt
726 uprev = integrator.u
727 end
728 # Handle Rosenbrock has no nlsolver so passes cache directly
729 cache = nlsolver isa OrdinaryDiffEqCache ? nlsolver : nlsolver.cache
730
731 isdae = integrator.alg isa DAEAlgorithm
732 if !isdae
733 mass_matrix = integrator.f.mass_matrix
734 end
735 isarray = uprev isa AbstractArray
736 # calculate W
737 is_compos = integrator.alg isa CompositeAlgorithm
738 islin, isode = islinearfunction(integrator)
739 !isdae && update_coefficients!(mass_matrix, uprev, p, t)
740
741 if cache.W isa AbstractSciMLOperator && !(cache.W isa Union{WOperator, StaticWOperator})
742 J = update_coefficients(cache.J, uprev, p, t)
743 W = update_coefficients(cache.W, uprev, p, t; dtgamma, transform = W_transform)
744 elseif islin
745 J = isode ? f.f : f.f1.f # unwrap the Jacobian accordingly
746 W = WOperator{false}(mass_matrix, dtgamma, J, uprev; transform = W_transform)
747 elseif DiffEqBase.has_jac(f)
748 J = f.jac(uprev, p, t)
749 if J isa StaticArray &&
750 integrator.alg isa
751 Union{Rosenbrock23, Rodas4, Rodas4P, Rodas4P2, Rodas5, Rodas5P}
752 W = W_transform ? J - mass_matrix * inv(dtgamma) :
753 dtgamma * J - mass_matrix
754 else
755 if !isa(J, AbstractSciMLOperator) && (!isnewton(nlsolver) ||
756 nlsolver.cache.W.J isa AbstractSciMLOperator)
757 J = MatrixOperator(J)
758 end
759 W = WOperator{false}(mass_matrix, dtgamma, J, uprev, cache.W.jacvec;
760 transform = W_transform)
761 end
762 integrator.stats.nw += 1
763 else
764 integrator.stats.nw += 1
765 J = calc_J(integrator, cache, next_step)
766 if isdae
767 W = J
768 else
769 W_full = W_transform ? J - mass_matrix * inv(dtgamma) :
770 dtgamma * J - mass_matrix
771 len = StaticArrayInterface.known_length(typeof(W_full))
772 W = if W_full isa Number
773 W_full
774 elseif len !== nothing &&
775 integrator.alg isa
776 Union{Rosenbrock23, Rodas4, Rodas4P, Rodas4P2, Rodas5, Rodas5P}
777 StaticWOperator(W_full)
778 else
779 DiffEqBase.default_factorize(W_full)
780 end
781 end
782 end
783 (W isa WOperator && unwrap_alg(integrator, true) isa NewtonAlgorithm) &&
784 (W = update_coefficients!(W, uprev, p, t)) # we will call `update_coefficients!` in NLNewton
785 is_compos && (integrator.eigen_est = isarray ? constvalue(opnorm(J, Inf)) :
786 integrator.opts.internalnorm(J, t))
787 return W
788 end

845 function build_J_W(alg, u, uprev, p, t, dt, f::F, ::Type{uEltypeNoUnits},
846 ::Val{IIP}) where {IIP, uEltypeNoUnits, F}
847 # TODO - make J, W AbstractSciMLOperators (lazily defined with scimlops functionality)
848 # TODO - if jvp given, make it SciMLOperators.FunctionOperator
849 # TODO - make mass matrix a SciMLOperator so it can be updated with time. Default to IdentityOperator
850 islin, isode = islinearfunction(f, alg)
851 if isdefined(f, :W_prototype) && (f.W_prototype isa AbstractSciMLOperator)
852 # We use W_prototype when it is provided as a SciMLOperator, and in this case we require jac_prototype to be a SciMLOperator too.
853 if !(f.jac_prototype isa AbstractSciMLOperator)
854 error("SciMLOperator for W_prototype only supported when jac_prototype is a SciMLOperator, but got $(typeof(f.jac_prototype))")
855 end
856 W = f.W_prototype
857 J = f.jac_prototype
858 elseif f.jac_prototype isa AbstractSciMLOperator
859 W = WOperator{IIP}(f, u, dt)
860 J = W.J
861 elseif IIP && f.jac_prototype !== nothing && concrete_jac(alg) === nothing &&
862 (alg.linsolve === nothing ||
863 alg.linsolve !== nothing &&
864 LinearSolve.needs_concrete_A(alg.linsolve))
865
866 # If factorization, then just use the jac_prototype
867 J = similar(f.jac_prototype)
868 W = similar(J)
869 elseif (IIP && (concrete_jac(alg) === nothing || !concrete_jac(alg)) &&
870 alg.linsolve !== nothing &&
871 !LinearSolve.needs_concrete_A(alg.linsolve))
872 # If the user has chosen GMRES but no sparse Jacobian, assume that the dense
873 # Jacobian is a bad idea and create a fully matrix-free solver. This can
874 # be overridden with concrete_jac.
875
876 _f = islin ? (isode ? f.f : f.f1.f) : f
877 jacvec = JacVec((du, u, p, t) -> _f(du, u, p, t), copy(u), p, t;
878 autodiff = alg_autodiff(alg), tag = OrdinaryDiffEqTag())
879 J = jacvec
880 W = WOperator{IIP}(f.mass_matrix, dt, J, u, jacvec)
881
882 elseif alg.linsolve !== nothing && !LinearSolve.needs_concrete_A(alg.linsolve) ||
883 concrete_jac(alg) !== nothing && concrete_jac(alg)
884 # The linear solver does not need a concrete Jacobian, but the user has
885 # asked for one. This will happen when the Jacobian is used in the preconditioner
886 # Thus setup JacVec and a concrete J, using sparsity when possible
887 _f = islin ? (isode ? f.f : f.f1.f) : f
888 J = if f.jac_prototype === nothing
889 ArrayInterface.undefmatrix(u)
890 else
891 deepcopy(f.jac_prototype)
892 end
893 __f = if IIP
894 (du, u, p, t) -> _f(du, u, p, t)
895 else
896 (u, p, t) -> _f(u, p, t)
897 end
898 jacvec = JacVec(__f, copy(u), p, t;
899 autodiff = alg_autodiff(alg), tag = OrdinaryDiffEqTag())
900 W = WOperator{IIP}(f.mass_matrix, dt, J, u, jacvec)
901
902 elseif islin || (!IIP && DiffEqBase.has_jac(f))
903 J = islin ? (isode ? f.f : f.f1.f) : f.jac(uprev, p, t) # unwrap the Jacobian accordingly
904 if !isa(J, AbstractSciMLOperator)
905 J = MatrixOperator(J)
906 end
907 W = WOperator{IIP}(f.mass_matrix, dt, J, u)
908 else
909 J = if f.jac_prototype === nothing
910 ArrayInterface.undefmatrix(u)
911 else
912 deepcopy(f.jac_prototype)
913 end
914 isdae = alg isa DAEAlgorithm
915 W = if isdae
916 J
917 elseif IIP
918 similar(J)
919 else
920 len = StaticArrayInterface.known_length(typeof(J))
921 if len !== nothing &&
922 alg isa
923 Union{Rosenbrock23, Rodas4, Rodas4P, Rodas4P2, Rodas5, Rodas5P}
924 StaticWOperator(J, false)
925 else
926 ArrayInterface.lu_instance(J)
927 end
928 end
929 end
930 return J, W
931 end
