513 """
514 ode_interpolation!(out,tval::Number,ts,timeseries,ks)
515
516 Get the value at tval where the solution is known at the
517 times ts (sorted), with values timeseries and derivatives ks
518 """
519 function ode_interpolation!(out, tval::Number, id::I, idxs, deriv::D, p,
520 continuity::Symbol = :left) where {I, D}
521 @unpack ts, timeseries, ks, f, cache, differential_vars = id
522 @inbounds tdir = sign(ts[end] - ts[1])
523
524 if continuity === :left
525 # we have i₋ = i₊ = 1 if tval = ts[1], i₊ = i₋ + 1 = lastindex(ts) if tval > ts[end],
526 # and otherwise i₋ and i₊ satisfy ts[i₋] < tval ≤ ts[i₊]
527 i₊ = min(lastindex(ts), _searchsortedfirst(ts, tval, 2, tdir > 0))
528 i₋ = i₊ > 1 ? i₊ - 1 : i₊
529 else
530 # we have i₋ = i₊ - 1 = 1 if tval < ts[1], i₊ = i₋ = lastindex(ts) if tval = ts[end],
531 # and otherwise i₋ and i₊ satisfy ts[i₋] ≤ tval < ts[i₊]
532 i₋ = max(1, _searchsortedlast(ts, tval, 1, tdir > 0))
533 i₊ = i₋ < lastindex(ts) ? i₋ + 1 : i₋
534 end
535 id.sensitivitymode && error(SENSITIVITY_INTERP_MESSAGE)
536
537 @inbounds begin
538 dt = ts[i₊] - ts[i₋]
539 Θ = iszero(dt) ? oneunit(tval) / oneunit(dt) : (tval - ts[i₋]) / dt
540
541 if cache isa (FunctionMapCache) || cache isa FunctionMapConstantCache
542 ode_interpolant!(out, Θ, dt, timeseries[i₋], timeseries[i₊], 0, cache, idxs,
543 deriv, differential_vars)
544 elseif !id.dense
545 linear_interpolant!(out, Θ, dt, timeseries[i₋], timeseries[i₊], idxs, deriv)
546 elseif cache isa CompositeCache
547 _ode_addsteps!(ks[i₊], ts[i₋], timeseries[i₋], timeseries[i₊], dt, f, p,
548 cache.caches[id.alg_choice[i₊]]) # update the kcurrent
549 ode_interpolant!(out, Θ, dt, timeseries[i₋], timeseries[i₊], ks[i₊],
550 cache.caches[id.alg_choice[i₊]], idxs, deriv, differential_vars)
551 else
552 _ode_addsteps!(ks[i₊], ts[i₋], timeseries[i₋], timeseries[i₊], dt, f, p,
553 cache) # update the kcurrent
554 ode_interpolant!(out, Θ, dt, timeseries[i₋], timeseries[i₊], ks[i₊], cache,
555 idxs, deriv, differential_vars)
556 end
557 end
