744 cp_type::CriticalPointType;

764

766 T = eltype(sys.grid.origin)
767

769 saddle_paths = [GradientPath[] for i in 1:Threads.nthreads()]

772 # if cp_type is ring_cp
773 direction = (cp_type == bond_cp ? forward_dir : backward_dir)

776 principal_dir = (cp_type == bond_cp ? 3 : 1)

779 n_done = Atomic(0)
780 n_total = length(saddle_cps) * 2
781
782 # Find the saddle paths in parallel

784 (cp_i, cp) = saddle_cps[i]
785 # Compute eigenvectors of the Hessian at the saddle point
786 hess = sys.hess(cp.r)
787 eigen_vecs = eigvecs(hess)
788 # Create two paths, one on either side of the saddle point
789 start_offset = copy(offset)
790 for offset_dir in [-1, 1]
791 # Compute the gradient path starting point, offset from the saddle point
792 # in the direction of the principal eigenvector
793 while norm(
794 sys.grad(
795 Vector{T}(
796 cp.r + eigen_vecs[:, principal_dir] .* start_offset * offset_dir

798 ),

802 end

805 )
806 Threads.@atomic n_done.x += 1
807 start_grad_norm = norm(sys.grad(start_pt))
808 @debug "Creating saddle path $(n_done.x) of $n_total, for CP $cp_type $i (cp $cp_i), offset_dir $offset_dir, starting grad norm $start_grad_norm"
809
810 # Create the gradient path
811 gp = create_gradient_path(
812 start_pt,
813 sys,
814 direction;

816 rho_cutoff=rho_cutoff,
817 grad_cutoff=grad_cutoff,
