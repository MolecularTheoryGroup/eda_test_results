228 function get_gha_level()
229 env_str = get(ENV, "RUNNER_DEBUG", "false")
230 # `tryparse(Bool, "1")` does not work on Julia 1.0, so we special-case that value
231 # (which is the only documented value for `RUNNER_DEBUG`, so any other case should be
232 # due to the user setting it manually somehow).
233 env_str == "1" && return Debug
234 # Fallback for all other values:
235 is_debug = something(tryparse(Bool, env_str), false)
236 return is_debug ? Debug : Info
237 end
