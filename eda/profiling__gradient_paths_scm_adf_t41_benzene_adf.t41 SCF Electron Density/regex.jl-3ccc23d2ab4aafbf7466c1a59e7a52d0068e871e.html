91 """
92 @r_str -> Regex
93
94 Construct a regex, such as `r"^[a-z]*\$"`, without interpolation and unescaping (except for
95 quotation mark `"` which still has to be escaped). The regex also accepts one or more flags,
96 listed after the ending quote, to change its behaviour:
97
98 - `i` enables case-insensitive matching
99 - `m` treats the `^` and `\$` tokens as matching the start and end of individual lines, as
100 opposed to the whole string.
101 - `s` allows the `.` modifier to match newlines.
102 - `x` enables "comment mode": whitespace is enabled except when escaped with `\\`, and `#`
103 is treated as starting a comment.
104 - `a` enables ASCII mode (disables `UTF` and `UCP` modes). By default `\\B`, `\\b`, `\\D`,
105 `\\d`, `\\S`, `\\s`, `\\W`, `\\w`, etc. match based on Unicode character properties. With
106 this option, these sequences only match ASCII characters. This includes `\\u` also, which
107 will emit the specified character value directly as a single byte, and not attempt to
108 encode it into UTF-8. Importantly, this option allows matching against invalid UTF-8
109 strings, by treating both matcher and target as simple bytes (as if they were ISO/IEC
110 8859-1 / Latin-1 bytes) instead of as character encodings. In this case, this option is
111 often combined with `s`. This option can be further refined by starting the pattern with
112 (*UCP) or (*UTF).
113
114 See [`Regex`](@ref) if interpolation is needed.
115
116 # Examples
117 ```jldoctest
118 julia> match(r"a+.*b+.*?d\$"ism, "Goodbye,\\nOh, angry,\\nBad world\\n")
119 RegexMatch("angry,\\nBad world")
120 ```
121 This regex has the first three flags enabled.
122 """

168 RegexMatch <: AbstractMatch
169
170 A type representing a single match to a [`Regex`](@ref) found in a string.
171 Typically created from the [`match`](@ref) function.
172
173 The `match` field stores the substring of the entire matched string.
174 The `captures` field stores the substrings for each capture group, indexed by number.
175 To index by capture group name, the entire match object should be indexed instead,
176 as shown in the examples.
177 The location of the start of the match is stored in the `offset` field.
178 The `offsets` field stores the locations of the start of each capture group,
179 with 0 denoting a group that was not captured.
180
181 This type can be used as an iterator over the capture groups of the `Regex`,
182 yielding the substrings captured in each group.
183 Because of this, the captures of a match can be destructured.
184 If a group was not captured, `nothing` will be yielded instead of a substring.
185
186 Methods that accept a `RegexMatch` object are defined for [`iterate`](@ref),
187 [`length`](@ref), [`eltype`](@ref), [`keys`](@ref keys(::RegexMatch)), [`haskey`](@ref), and
188 [`getindex`](@ref), where keys are the the names or numbers of a capture group.
189 See [`keys`](@ref keys(::RegexMatch)) for more information.
190
191 # Examples

193 julia> m = match(r"(?<hour>\\d+):(?<minute>\\d+)(am|pm)?", "11:30 in the morning")
194 RegexMatch("11:30", hour="11", minute="30", 3=nothing)
195
196 julia> m.match
197 "11:30"
198
199 julia> m.captures
200 3-element Vector{Union{Nothing, SubString{String}}}:
201 "11"
202 "30"
203 nothing

205
206 julia> m["minute"]

209 julia> hr, min, ampm = m; # destructure capture groups by iteration

513 function count(t::Union{AbstractChar,AbstractString,AbstractPattern}, s::AbstractString; overlap::Bool=false)
514 n = 0
515 i, e = firstindex(s), lastindex(s)
516 while true
517 r = findnext(t, s, i)
518 isnothing(r) && break
519 n += 1
520 j = overlap || isempty(r) ? first(r) : last(r)
521 j > e && break
522 @inbounds i = nextind(s, j)
523 end

653 if all(isdigit, groupname)
654 group = parse(Int, groupname)
655 elseif re isa RegexAndMatchData
656 group = PCRE.substring_number_from_name(re.re.regex, groupname)
657 group < 0 && replace_err("Group $groupname not found in regex $(re.re)")
658 else
659 group = -1
660 end
661 _write_capture(io, group, str, r, re)
662 i = nextind(repl, i)
663 else
664 replace_err(repl)
665 end
666 else
667 write(io, repl[i])
668 i = nextind(repl, i)
669 end
670 end
671 end

687 opts_nonempty = UInt32(PCRE.ANCHORED | PCRE.NOTEMPTY_ATSTART)
688 while true
689 mat = match(itr.regex, itr.string, offset,
690 prevempty ? opts_nonempty : UInt32(0))
691
692 if mat === nothing
693 if prevempty && offset <= sizeof(itr.string)
694 offset = nextind(itr.string, offset)
695 prevempty = false
696 continue
697 else
698 break
699 end
700 else
701 if itr.overlap
702 if !isempty(mat.match)
703 offset = nextind(itr.string, mat.offset)
704 else
705 offset = mat.offset
706 end
707 else
708 offset = mat.offset + ncodeunits(mat.match)
709 end
710 return (mat, (offset, isempty(mat.match)))
711 end
712 end
713 nothing

763 """
764 *(s::Regex, t::Union{Regex,AbstractString,AbstractChar}) -> Regex
765 *(s::Union{Regex,AbstractString,AbstractChar}, t::Regex) -> Regex
766
767 Concatenate regexes, strings and/or characters, producing a [`Regex`](@ref).
768 String and character arguments must be matched exactly in the resulting regex,
769 meaning that the contained characters are devoid of any special meaning
770 (they are quoted with "\\Q" and "\\E").
771
772 !!! compat "Julia 1.3"
773 This method requires at least Julia 1.3.
774
775 # Examples
776 ```jldoctest
777 julia> match(r"Hello|Good bye" * ' ' * "world", "Hello world")
778 RegexMatch("Hello world")
779
780 julia> r = r"a|b" * "c|d"
781 r"(?:a|b)\\Qc|d\\E"
782
783 julia> match(r, "ac") == nothing
784 true
785
786 julia> match(r, "ac|d")
787 RegexMatch("ac|d")
788 ```
789 """
790 function *(r1::Union{Regex,AbstractString,AbstractChar}, rs::Union{Regex,AbstractString,AbstractChar}...)
791 mask = PCRE.CASELESS | PCRE.MULTILINE | PCRE.DOTALL | PCRE.EXTENDED # imsx
792 match_opts = nothing # all args must agree on this
793 compile_opts = nothing # all args must agree on this
794 shared = mask
795 for r in (r1, rs...)
796 r isa Regex || continue
797 if match_opts === nothing
798 match_opts = r.match_options
799 compile_opts = r.compile_options & ~mask
800 else
801 r.match_options == match_opts &&
802 r.compile_options & ~mask == compile_opts ||
803 throw(ArgumentError("cannot multiply regexes: incompatible options"))
804 end
805 shared &= r.compile_options
806 end
807 unshared = mask & ~shared
808 Regex(string(wrap_string(r1, unshared), wrap_string.(rs, Ref(unshared))...), compile_opts | shared, match_opts)
809 end
810
811 *(r::Regex) = r # avoids wrapping r in a useless subpattern
812
813 wrap_string(r::Regex, unshared::UInt32) = string("(?", regex_opts_str(r.compile_options & unshared), ':', r.pattern, ')')
814 # if s contains raw"\E", split '\' and 'E' within two distinct \Q...\E groups:
815 wrap_string(s::AbstractString, ::UInt32) = string("\\Q", replace(s, raw"\E" => raw"\\E\QE"), "\\E")
816 wrap_string(s::AbstractChar, ::UInt32) = string("\\Q", s, "\\E")
817
818 regex_opts_str(opts) = (isassigned(_regex_opts_str) ? _regex_opts_str[] : init_regex())[opts]
819
820 # UInt32 to String mapping for some compile options
821 const _regex_opts_str = Ref{ImmutableDict{UInt32,String}}()
822
823 @noinline init_regex() = _regex_opts_str[] = foldl(0:15, init=ImmutableDict{UInt32,String}()) do d, o
824 opt = UInt32(0)
825 str = ""
826 if o & 1 != 0
827 opt |= PCRE.CASELESS
828 str *= 'i'
829 end
830 if o & 2 != 0
831 opt |= PCRE.MULTILINE
832 str *= 'm'
833 end
834 if o & 4 != 0
835 opt |= PCRE.DOTALL
836 str *= 's'
837 end
838 if o & 8 != 0
839 opt |= PCRE.EXTENDED
840 str *= 'x'
841 end
842 ImmutableDict(d, opt => str)
843 end
