236 @inline function _compute_rhs(nlsolver::NLSolver{<:NLNewton, false}, integrator, f::TF,
237 z) where {TF}
238 @unpack uprev, t, p, dt = integrator
239 @unpack tmp, ztmp, γ, α, cache = nlsolver
240 @unpack tstep, invγdt = cache
241 isdae = TF <: DAEFunction
242 if isdae
243 # not all predictors are uprev, for other forms of predictors, defined in u₀
244 if isdefined(integrator.cache, :u₀)
245 ustep = @.. broadcast=false integrator.cache.u₀+z
246 else
247 ustep = @.. broadcast=false uprev+z
248 end
249 dustep = @. (tmp + α * z) * invγdt
250 ztmp = f(dustep, ustep, p, t)
251 else
252 mass_matrix = integrator.f.mass_matrix
253 if nlsolver.method === COEFFICIENT_MULTISTEP
254 ustep = z
255 # tmp = outertmp ./ hγ
256 if mass_matrix === I
257 ztmp = tmp .+ f(z, p, tstep) .- (α * invγdt) .* z
258 else
259 update_coefficients!(mass_matrix, ustep, p, tstep)
260 ztmp = tmp .+ f(z, p, tstep) .- (mass_matrix * z) .* (α * invγdt)
261 end
262 else
263 ustep = @. tmp + γ * z
264 if mass_matrix === I
265 ztmp = (dt * f(ustep, p, tstep) - z) * invγdt
266 else
267 update_coefficients!(mass_matrix, ustep, p, tstep)
268 ztmp = (dt * f(ustep, p, tstep) - mass_matrix * z) * invγdt
269 end
270 end
271 end
