259
260 Compute the principal curvatures and directions of an isosurface at a given point.

271 r = [1.0, 2.0, 3.0]

273 curvatures, directions = principal_curvatures_and_directions(r, sys)

275 """
276 function principal_curvatures_and_directions(
277 r::Vector{T}, sys::System
278 )::Tuple{Vector{T},Matrix{T}} where {T<:AbstractFloat}
279 # Compute gradient and hessian at r
280 grad_rho = sys.grad(r)
281 hessian_rho = sys.hess(r)
282
283 grad_norm = norm(grad_rho)
284 N = grad_rho ./ grad_norm
285
286 # make unit diagonal whose size matches grad_rho
287 D = Diagonal(fill(1.0, length(grad_rho)))
288
289 P = D - N * N'
290
291 # Compute the shape operator
292 S = P * (hessian_rho ./ grad_norm)
293
294 # Compute the eigenvalues and eigenvectors of the shape operator
295 e = eigen(S)
296
297 # Sort indices based on abssolute value of eigen values
298 sorted_indices = sortperm(1:3; by=i -> -abs(e.values[i]))
299
300 # In the 3d case, k1 and k2 are the greatest and second greatest absolute value eigenvalues,
301 # and are stored in vals.
302 # v1 and v2 are the corresponding eigenvectors, and are stored in vecs.
303 # This must be generalized for any dimensionality D, so vals will be the
304 # D-1 greatest eigenvalues, indescending order, and vecs will be the corresponding eigenvectors.
305
306 d = length(grad_rho) - 1
307 vals = Vector{T}(-e.values[sorted_indices[1:d]])
308 vecs = Matrix{T}(e.vectors[:, sorted_indices[1:d]])
309
310 return vals, vecs
311 end
